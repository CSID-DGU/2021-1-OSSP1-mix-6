#include <iostream>
#include <time.h>

int main(){
    std::cout << "Hello World!\n";
    clock_t start, stop;
    double duration;
    start = clock();
    for (int i = 0; i < 100; i++) {
        std::cout << i << "를 출력합니다";
    }
    stop = clock();
    duration = (double)(stop - start) / CLOCKS_PER_SEC;
    std::cout << "수행시간은 " << duration << "초입니다.";
}
